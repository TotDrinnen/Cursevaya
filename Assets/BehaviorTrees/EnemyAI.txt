tree("Root")
		fallback		
			tree("Fight")
			tree("Patrol")
	
tree("Patrol")
	while
		sequence
			not
				isSeePlayer
		sequence
			StartPatrol	
			repeat
				sequence
					isPatroller
					SetWayPoint
					MoveToNextWayPoint
					Wait(0.3)
					SetIndex
tree("Fight")
	sequence
		isAlive
		isSeePlayer
		Wait(0.6)
		parallel
			MoveToEnemy
			repeat
				Attack
